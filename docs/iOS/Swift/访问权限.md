# 访问权限

open：允许在任意的地方访问

public

internal：默认的级别，当前文件可以访问

fliePrivate：只能在定义的源文件访问

private：只能在定义的类内部使用

```swift
// MARK: private
class LGPersonCS{
    private var sex: Bool
    
    private func unpdateSex(){
        self.sex = !self.sex
    }
    
    init(sex innerSex: Bool) {
        self.sex = innerSex
    }
    
    func test(){
        self.unpdateSex()
    }
}

func testP() {
    let p = LGPersonCS(sex: true)
    p.test()
}
```
