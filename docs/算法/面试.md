## 去除重复字符

```objective-c
//例如：fabccdeeffacb打印fabcde 每个字符打印一次
- (void)test1 {
    NSString *str = @"fabccdeeffacb";
    
    NSMutableArray *array = [NSMutableArray arrayWithCapacity:0];
    NSMutableString *tempStr = [NSMutableString stringWithCapacity:0];
    for (int i = 0; i < str.length; i++) {
        //不包含
        if (![array containsObject:[str substringWithRange:NSMakeRange(i, 1)]]) {
          //
          [array addObject:[str substringWithRange:NSMakeRange(i, 1)]];
          [tempStr appendString:[str substringWithRange:NSMakeRange(i, 1)]];
        }
    }
    
    NSLog(@"%@",tempStr);
}
```

## 从一个数组中找出重复的元素打印出来

```objective-c
//从一个数组中找出重复的元素打印出来
- (void)test2
{
/**
 打印结果：
 (
     1,
     2,
     5
 )
 */
    NSArray *arr = [NSArray arrayWithObjects:@"1",@"2",@"1",@"7",@"4",@"5",@"2", @"6",@"5",nil];
    NSMutableArray *arrmu = [[NSMutableArray alloc]init];//遍历时未出现重复的 保存在这里
    NSMutableArray *sameArray = [[NSMutableArray alloc]init];//遍历时已经有重复的 保存在这里
    for (int i = 0 ; i < [arr count]; i++) {
        id str = [arr objectAtIndex:i];
        
        //如果之前已经重复了。就不再保存在sameArray中了
        BOOL sameArrayHasContain = NO;
        for (int sameArrayIndex = 0; sameArrayIndex < sameArray.count; sameArrayIndex++) {
            if ([sameArray[sameArrayIndex] isEqual:str]) {
                sameArrayHasContain = YES;
                break;
            }
        }
        if (sameArrayHasContain) {
            break;
        }
        
      
        if ([arrmu count] == 0)
        {
            [arrmu addObject:str];
        } else {
            BOOL flag = NO;
            for (int j = 0; j < [arrmu count]; j++ ) {
                if ([str isEqual:[arrmu objectAtIndex:j]]) {
                    [sameArray addObject:str];
                    flag =YES;
                    break;
                }
            }
            if (flag == NO) {
                [arrmu addObject:str];
            }
        }
    }
  
    NSLog(@"sameArray : %@",sameArray);
}
```
