    CFRunLoopRef CFRunLoopGetMain(void) {
    CHECK_FOR_FORK();
    static CFRunLoopRef __main = NULL; // no retain needed

}

    return _CFRunLoopGet0(pthread_self());

    CFRunLoopRef mainLoop = __CFRunLoopCreate(pthread_main_thread_np());

    // иҝӣиЎҢз»‘е®ҡ dict[@"pthread_main_thread_np"] = mainLoop
    //key - value зҡ„еҪўејҸеӯҳж”ҫпјҢдё»зәҝзЁӢз»‘е®ҡдё»runloop  
    CFDictionarySetValue(dict, pthreadPointer(pthread_main_thread_np()), mainLoop);

    if (!OSAtomicCompareAndSwapPtrBarrier(NULL, dict, (void * volatile *)&__CFRunLoops)) {
        CFRelease(dict);
    }
    CFRelease(mainLoop);
        __CFLock(&loopsLock);
    }
    //йҖҡиҝҮзәҝзЁӢзӣҙжҺҘд»ҺdictдёӯиҺ·еҸ–loop
    CFRunLoopRef loop = (CFRunLoopRef)CFDictionaryGetValue(__CFRunLoops, pthreadPointer(t));
    __CFUnlock(&loopsLock);
    //еҰӮжһңжІЎжңү еҲҷйҖҡиҝҮзәҝзЁӢеҲӣе»әдёҖдёӘloop
    if (!loop) {
            CFRunLoopRef newLoop = __CFRunLoopCreate(t);
        __CFLock(&loopsLock);
          loop = (CFRunLoopRef)CFDictionaryGetValue(__CFRunLoops, pthreadPointer(t));
    if (!loop) {
        //еҶҚж¬ЎзЎ®и®ӨжІЎжңүloopпјҢеҲҷж·»еҠ еҲ°е…ЁеұҖеҸҜеҸҳеӯ—е…ёдёӯ
        CFDictionarySetValue(__CFRunLoops, pthreadPointer(t), newLoop);
        loop = newLoop;
    }
        // don't release run loops inside the loopsLock, because CFRunLoopDeallocate may end up taking it
        __CFUnlock(&loopsLock);
          CFRelease(newLoop);
    }
    if (pthread_equal(t, pthread_self())) {
          //жіЁеҶҢдёҖдёӘеӣһи°ғпјҢеҪ“зәҝзЁӢй”ҖжҜҒж—¶пјҢйЎәдҫҝд№ҹй”ҖжҜҒе…¶еҜ№еә”зҡ„RunloopгҖӮ
        _CFSetTSD(__CFTSDKeyRunLoop, (void *)loop, NULL);
        if (0 == _CFGetTSD(__CFTSDKeyRunLoopCntr)) {
            _CFSetTSD(__CFTSDKeyRunLoopCntr, (void *)(PTHREAD_DESTRUCTOR_ITERATIONS-1), (void (*)(void *))__CFFinalizeRunLoop);
        }
    }
    return loop;
}

    зәҝзЁӢе’ҢrunloopжҳҜдёҖдёҖз»‘е®ҡзҡ„е…ізі»гҖӮ¶

    RunLoopеҜ№иұЎжҳҜеҲ©з”Ёеӯ—е…ёжқҘиҝӣиЎҢеӯҳеӮЁпјҢиҖҢдё”keyеҜ№еә”зҡ„зәҝзЁӢValueдёәиҜҘзәҝзЁӢеҜ№еә”зҡ„RunLoopгҖӮ

    еҸӘиғҪйҖҡиҝҮCFRunLoopGetMainеҮҪж•°жҲ–иҖ…CFRunLoopGetCurrentеҮҪж•°жқҘиҺ·еҸ–RunLoopпјҢж— и®әжҳҜCFRunLoopGetMainеҮҪж•°иҝҳжҳҜCFRunLoopGetCurrentеҮҪж•°пјҢйғҪжҳҜйҖҡиҝҮеҜ№еә”зҡ„зәҝзЁӢиҺ·еҸ–еҜ№еә”зҡ„RunLoopпјҢзәҝзЁӢе’ҢRunLoopжҳҜдёҖдёҖеҜ№еә”зҡ„пјҢдёҚдјҡйҮҚеӨҚеҲӣе»әгҖӮ

    дё»зәҝзЁӢзҡ„runloopжҳҜй»ҳи®ӨеҲӣе»әе’ҢејҖеҗҜзҡ„пјҢеӯҗзәҝзЁӢзҡ„runloopйңҖиҰҒжүӢеҠЁеҲӣе»әе’ҢејҖеҗҜгҖӮ¶

    ејҖдёҖдёӘеӯҗзәҝзЁӢеҲӣе»әrunloopпјҢдёҚжҳҜйҖҡиҝҮalloc initж–№жі•еҲӣе»әпјҢиҖҢжҳҜзӣҙжҺҘйҖҡиҝҮи°ғз”ЁcurrentRunLoopж–№жі•жқҘеҲӣе»әпјҢе®ғжң¬иә«жҳҜдёҖдёӘжҮ’еҠ иҪҪзҡ„гҖӮ

    еңЁеӯҗзәҝзЁӢдёӯпјҢеҰӮжһңдёҚдё»еҠЁиҺ·еҸ–RunLoopзҡ„иҜқпјҢйӮЈд№ҲеӯҗзәҝзЁӢеҶ…йғЁжҳҜдёҚдјҡеҲӣе»әRunLoopзҡ„гҖӮ

    еңЁдё»зәҝзЁӢпјҢзі»з»ҹдјҡеё®жҲ‘д»¬еҲӣе»әRunLoopпјҢжқҘеӨ„зҗҶдәӢд»¶гҖӮиҖҢеӯҗзәҝзЁӢRunLoopе№¶дёҚдјҡй»ҳи®ӨејҖеҗҜгҖӮ

    еӯҗзәҝзЁӢж“ҚдҪңе®ҢжҲҗеҗҺпјҢзәҝзЁӢе°ұиў«й”ҖжҜҒдәҶпјҢеҰӮжһңжҲ‘д»¬жғізәҝзЁӢдёҚиў«й”ҖжҜҒпјҢеҫ—дё»еҠЁиҺ·еҸ–дёҖдёӘRunLoopпјҢе№¶дё”еңЁRunLoopдёӯж·»еҠ Timer/Source/Observerе…¶дёӯзҡ„дёҖдёӘгҖӮ

    static CFRunLoopRef __CFRunLoopCreate(pthread_t t) {
    CFRunLoopRef loop = NULL;
    CFRunLoopModeRef rlm;
    uint32_t size = sizeof(struct __CFRunLoop) - sizeof(CFRuntimeBase);
    loop = (CFRunLoopRef)_CFRuntimeCreateInstance(kCFAllocatorSystemDefault, CFRunLoopGetTypeID(), size, NULL);
    if (NULL == loop) {
        return NULL;
    }
    (void)__CFRunLoopPushPerRunData(loop);
    __CFRunLoopLockInit(&loop->_lock);
    loop->_wakeUpPort = __CFPortAllocate();
    if (CFPORT_NULL == loop->_wakeUpPort) HALT;
    __CFRunLoopSetIgnoreWakeUps(loop);
    loop->_commonModes = CFSetCreateMutable(kCFAllocatorSystemDefault, 0, &kCFTypeSetCallBacks);//setзұ»еһӢзҡ„commonModes ж— еәҸзҡ„йӣҶеҗҲ
    CFSetAddValue(loop->_commonModes, kCFRunLoopDefaultMode);
    loop->_commonModeItems = NULL;
    loop->_currentMode = NULL;
    loop->_modes = CFSetCreateMutable(kCFAllocatorSystemDefault, 0, &kCFTypeSetCallBacks);
    loop->_blocks_head = NULL;//еӨҙ
    loop->_blocks_tail = NULL;//е°ҫ
    loop->_counterpart = NULL;
    loop->_pthread = t;
#if DEPLOYMENT_TARGET_WINDOWS
    loop->_winthread = GetCurrentThreadId();
#else
    loop->_winthread = 0;
#endif
    rlm = __CFRunLoopFindMode(loop, kCFRunLoopDefaultMode, true);
    if (NULL != rlm) __CFRunLoopModeUnlock(rlm);
    return loop;
}
