    weakејұеј•вҪӨ¶

    ејұеј•вҪӨдёҚдјҡеҜ№е…¶еј•вҪӨзҡ„е®һдҫӢдҝқжҢҒејәеј•вҪӨпјҢеӣ вҪҪдёҚдјҡйҳ»вҪҢ ARC йҮҠж”ҫиў«еј•вҪӨзҡ„е®һдҫӢгҖӮиҝҷдёӘзү№жҖ§йҳ»вҪҢдәҶеј•вҪӨеҸҳдёәеҫӘзҺҜејәеј•вҪӨгҖӮеЈ°жҳҺеұһжҖ§жҲ–иҖ…еҸҳйҮҸж—¶пјҢеңЁеүҚвҫҜеҠ дёҠweakе…ій”®еӯ—иЎЁжҳҺиҝҷжҳҜвјҖдёӘејұеј•вҪӨгҖӮ

    з”ұдәҺејұеј•вҪӨдёҚдјҡејәдҝқжҢҒеҜ№е®һдҫӢзҡ„еј•вҪӨпјҢжүҖд»ҘиҜҙе®һдҫӢиў«йҮҠж”ҫдәҶејұеј•вҪӨд»Қж—§еј•вҪӨзқҖиҝҷдёӘе®һдҫӢд№ҹжҳҜжңүеҸҜиғҪзҡ„гҖӮеӣ жӯӨпјҢARC дјҡеңЁиў«еј•вҪӨзҡ„е®һдҫӢиў«йҮҠж”ҫжҳҜвҫғеҠЁең°и®ҫзҪ®ејұеј•вҪӨдёә nil гҖӮз”ұдәҺејұеј•вҪӨйңҖиҰҒе…Ғи®ёе®ғд»¬зҡ„еҖјдёә nil пјҢе®ғд»¬вјҖе®ҡеҫ—жҳҜеҸҜйҖүзұ»еһӢгҖӮ

    • ејұеј•з”ЁдёҚдјҡеўһеҠ е®һдҫӢзҡ„еј•з”Ёи®Ўж•°пјҢеӣ жӯӨдёҚдјҡйҳ»жӯўARCй”ҖжҜҒиў«еј•з”Ёзҡ„е®һдҫӢгҖӮиҝҷз§Қзү№жҖ§дҪҝеҫ—еј•з”ЁдёҚдјҡеҸҳжҲҗејәеј•з”ЁзҺҜгҖӮеЈ°жҳҺеұһжҖ§жҲ–иҖ…еҸҳйҮҸзҡ„ж—¶еҖҷпјҢе…ій”®еӯ—weakиЎЁжҳҺеј•з”Ёдёәејұеј•з”ЁгҖӮ
    • ејұеј•з”ЁеҸӘиғҪеЈ°жҳҺдёәеҸҳйҮҸзұ»еһӢпјҢеӣ дёәиҝҗиЎҢж—¶е®ғзҡ„еҖјеҸҜиғҪж”№еҸҳгҖӮејұеј•з”Ёз»қеҜ№дёҚиғҪеЈ°жҳҺдёәеёёйҮҸгҖӮ
    • еӣ дёәејұеј•з”ЁеҸҜд»ҘжІЎжңүеҖјпјҢжүҖд»ҘеЈ°жҳҺејұеј•з”Ёзҡ„ж—¶еҖҷеҝ…йЎ»жҳҜеҸҜйҖүзұ»еһӢзҡ„гҖӮеңЁSwiftиҜӯиЁҖдёӯпјҢжҺЁиҚҗз”ЁеҸҜйҖүзұ»еһӢжқҘдҪңдёәеҸҜиғҪжІЎжңүеҖјзҡ„еј•з”Ёзҡ„зұ»еһӢгҖӮ

    unownedж— дё»еј•з”Ё¶

    • е’Ңејұеј•з”ЁзӣёдјјпјҢж— дё»еј•з”Ёд№ҹдёҚејәжҢҒжңүе®һдҫӢгҖӮдҪҶжҳҜе’Ңејұеј•з”ЁдёҚеҗҢзҡ„жҳҜпјҢж— дё»еј•з”Ёй»ҳи®Өе§Ӣз»ҲжңүеҖјгҖӮеӣ жӯӨпјҢж— дё»еј•з”ЁеҸӘиғҪе®ҡд№үдёәйқһеҸҜйҖүзұ»еһӢпјҲnon-optional typeпјүгҖӮеңЁеұһжҖ§гҖҒеҸҳйҮҸеүҚж·»еҠ  unowned е…ій”®еӯ—пјҢеҸҜд»ҘеЈ°жҳҺдёҖдёӘж— дё»еј•з”ЁгҖӮ
    • еӣ дёәжҳҜйқһеҸҜйҖүзұ»еһӢпјҢеӣ жӯӨеҪ“дҪҝз”Ёж— дё»еј•з”Ёзҡ„ж—¶еҖҷпјҢдёҚйңҖиҰҒеұ•ејҖпјҢеҸҜд»ҘзӣҙжҺҘи®ҝй—®гҖӮдёҚиҝҮйқһеҸҜйҖүзұ»еһӢеҸҳйҮҸдёҚиғҪиөӢеҖјдёә nilпјҢеӣ жӯӨеҪ“е®һдҫӢиў«й”ҖжҜҒзҡ„ж—¶еҖҷпјҢARC ж— жі•е°Ҷеј•з”ЁиөӢеҖјдёә nilгҖӮ
    • еҪ“е®һдҫӢиў«й”ҖжҜҒеҗҺпјҢиҜ•еӣҫи®ҝй—®иҜҘе®һдҫӢзҡ„ж— дё»еј•з”Ёдјҡи§ҰеҸ‘иҝҗиЎҢж—¶й”ҷиҜҜгҖӮдҪҝз”Ёж— дё»еј•з”Ёж—¶иҜ·зЎ®дҝқеј•з”Ёе§Ӣз»ҲжҢҮеҗ‘дёҖдёӘжңӘй”ҖжҜҒзҡ„е®һдҫӢгҖӮ

    ж №жҚ®иӢ№жһңзҡ„е®ҳвҪ…вҪӮжЎЈзҡ„е»әи®®гҖӮеҪ“дёӨдёӘеҜ№иұЎзҡ„вҪЈе‘Ҫе‘Ёжңҹе№¶дёҚзӣёе…іпјҢйӮЈд№ҲжҲ‘д»¬еҝ…йЎ»дҪҝвҪӨ weakгҖӮзӣёеҸҚпјҢвҫ®ејәеј•вҪӨеҜ№иұЎжӢҘжңүе’Ңејәеј•вҪӨеҜ№иұЎеҗҢж ·жҲ–иҖ…жӣҙв»“зҡ„вҪЈе‘Ҫе‘Ёжңҹзҡ„иҜқпјҢеҲҷеә”иҜҘдҪҝвҪӨ unownedгҖӮ

    weak VS unowned¶

    • еҰӮжһңдёӨдёӘеҜ№иұЎзҡ„вҪЈе‘Ҫе‘Ёжңҹе®Ңе…Ёе’ҢеҜ№вҪ…жІЎе…ізі»пјҲе…¶дёӯвјҖвҪ…д»Җд№Ҳж—¶еҖҷиөӢеҖјдёәnilпјҢеҜ№еҜ№вҪ…йғҪжІЎеҪұе“ҚпјүпјҢиҜ·вҪӨ weakпјҲдҫӢdelegateпјүгҖӮ
    • еҰӮжһңдҪ зҡ„д»Јз ҒиғҪзЎ®дҝқпјҡе…¶дёӯвјҖдёӘеҜ№иұЎй”ҖжҜҒпјҢеҸҰвјҖдёӘеҜ№иұЎд№ҹиҰҒи·ҹзқҖй”ҖжҜҒпјҢиҝҷж—¶еҖҷпјҢеҸҜд»ҘпјҲи°Ёж…ҺпјүвҪӨ unownedгҖӮ

    еҰӮжһңдёҚзЎ®е®ҡзҡ„иҜқпјҢзӣҙжҺҘдҪҝз”ЁweakгҖӮunownedжҖ§иғҪзЁҚеҘҪпјҢзӣҙжҺҘж“ҚдҪңдҪҚеҹҹдҝЎжҒҜгҖӮweakдјҡеҲӣе»әsideиЎЁпјҢеҜ№ж•ЈеҲ—иЎЁж“ҚдҪңгҖӮ

    weakдҝ®йҘ°зҡ„жҳҜдёҖдёӘеҸҜйҖүзұ»еһӢпјҢunownedдёҚжҳҜеҸҜйҖүзұ»еһӢпјҢеҒҮе®ҡжңүеҖјгҖӮжүҖд»Ҙweakе®үе…ЁдёҖзӮ№гҖӮ

    вҖў ејұеј•з”Ёе’Ңж— дё»еј•з”Ёе…Ғи®ёеј•з”ЁзҺҜдёӯзҡ„дёҖдёӘе®һдҫӢеј•з”ЁеҸҰеӨ–дёҖдёӘе®һдҫӢпјҢдҪҶдёҚжҳҜејәеј•з”ЁгҖӮеӣ жӯӨе®һдҫӢеҸҜд»Ҙдә’зӣёеј•з”ЁпјҢдҪҶжҳҜдёҚдјҡдә§з”ҹејәеј•з”ЁзҺҜгҖӮ

    вҖў еҜ№дәҺз”ҹе‘Ҫе‘Ёжңҹдёӯеј•з”ЁдјҡеҸҳдёә nil зҡ„е®һдҫӢпјҢдҪҝз”Ёејұеј•з”ЁпјӣеҜ№дәҺеҲқе§ӢеҢ–ж—¶иөӢеҖјд№ӢеҗҺеј•з”ЁеҶҚд№ҹдёҚдјҡиөӢеҖјдёә nil

    зҡ„е®һдҫӢпјҢдҪҝз”Ёж— дё»еј•з”ЁгҖӮ

    й—ӯеҢ…еҫӘзҺҜеј•з”Ё¶

    е°ҶдёҖдёӘй—ӯеҢ…иөӢеҖјз»ҷзұ»е®һдҫӢзҡ„жҹҗдёӘеұһжҖ§пјҢе№¶дё”иҝҷдёӘй—ӯеҢ…дҪҝз”ЁдәҶе®һдҫӢпјҢиҝҷж ·д№ҹдјҡдә§з”ҹејәеј•з”ЁзҺҜгҖӮиҝҷдёӘй—ӯеҢ…еҸҜиғҪи®ҝй—®дәҶе®һдҫӢзҡ„жҹҗдёӘеұһжҖ§пјҢдҫӢеҰӮ self.somePropertyпјҢжҲ–иҖ…и°ғз”ЁдәҶе®һдҫӢзҡ„жҹҗдёӘж–№жі•пјҢдҫӢеҰӮself.someMethodгҖӮиҝҷдёӨз§Қжғ…еҶөйғҪеҜјиҮҙдәҶй—ӯеҢ…дҪҝз”Ё selfпјҢд»ҺиҖҢдә§з”ҹдәҶеҫӘзҺҜеј•з”ЁгҖӮ

    вҫёе…ҲжҲ‘д»¬зҡ„й—ӯеҢ…дјҡвјҖиҲ¬й»ҳи®ӨжҚ•иҺ·жҲ‘д»¬еӨ–йғЁзҡ„еҸҳйҮҸ 

    var age = 18

let closure = {
  age += 1
}

    д»Һжү“еҚ°з»“жһң19еҸҜд»ҘзңӢеҮәжқҘ

    й—ӯеҢ…еҶ…йғЁеҜ№еҸҳйҮҸзҡ„дҝ®ж”№е°Ҷдјҡж”№еҸҳеӨ–йғЁеҺҹе§ӢеҸҳйҮҸзҡ„еҖј

    йӮЈеҗҢж ·е°ұдјҡжңүвјҖдёӘй—®йўҳпјҢеҰӮжһңеңЁ class зҡ„еҶ…йғЁе®ҡд№үвјҖдёӘй—ӯеҢ…пјҢеҪ“еүҚй—ӯеҢ…и®ҝй—®еұһжҖ§зҡ„иҝҮзЁӢдёӯпјҢе°ұдјҡеҜ№еҪ“еүҚзҡ„е®һдҫӢеҜ№иұЎиҝӣвҫҸжҚ•иҺ·пјҡ

    class HHTeacherC{
    var age = 18

    var closure:(()->())?

    //йҖҡиҝҮжү“еҚ°deinitжқҘжҹҘзңӢеҪ“еүҚеҜ№иұЎжҳҜеҗҰйҮҠж”ҫ
    deinit {
        print("HHTeacher deinit")
    }
}
func closureFunc() {
    let t = HHTeacherC()
    t.closure = {
        t.age += 1
    }
}

    class HHTeacherC{
    var age = 18

    var closure:(()->())?

    //йҖҡиҝҮжү“еҚ°deinitжқҘжҹҘзңӢеҪ“еүҚеҜ№иұЎжҳҜеҗҰйҮҠж”ҫ
    deinit {
        print("HHTeacher deinit")
    }
}
func closureFunc() {
    let t = HHTeacherC()

    //дҪҝз”Ёweak
    t.closure = { [weak t] in
        t!.age += 1

        // MARK: weak strong dance
        //зҪ‘з»ңиҜ·жұӮж—¶
        if let strongSelf = t{//еҜ№еҸҜйҖүеҖјtз»‘е®ҡ и§Јз»‘з»ҷstrongSelf
            print(strongSelf.age)
        }
        //жҲ–иҖ… е»¶й•ҝtзҡ„з”ҹе‘Ҫе‘ЁжңҹпјҲеңЁй—ӯеҢ…иЎЁиҫҫејҸt.closureзҡ„иҢғеӣҙпјү
        withExtendedLifetime(t){
            t!.age
        }
    }

    //дҪҝз”Ёunowned
    t.closure = { [unowned t] in
        t.age += 1
    }
}
